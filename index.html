Project: Binary Tree

บทคัดย่อ

โปรเจกต์เกม Binary Tree ทำขึ้นเพื่อใช้สำหรับรายวิชา Physical Computing ซึ่งเป็นเกมที่ออกแบบมาเพื่อใช้เซ็นเซอร์วัดอุณหภูมิ (Temperature Sensor) ในการตรวจจับอุณหภูมิและแสดงผลการเปลี่ยนแปลงของอุณหภูมิภายในเกม ในเกมจะมีต้นไม้ซึ่งพัฒนาการของต้นไม้ขึ้นอยู่กับอุณหภูมิที่เซ็นเซอร์ตรวจจับได้

#ifndef BRANCH_H
#define BRANCH_H

#include "raylib.h" // Needed for [Vector2] type
#include  // Needed for [bool] type

typedef struct {
    short animal_max;
} Building;

typedef struct {
    int id;
    Vector2 pos;
} Product;

struct Branch;
typedef struct Branch {
    int id;

    Vector2 pos;

    struct Branch* parent;
    struct Branch* child_first;
    struct Branch* child_second;

    bool is_leaves;
    bool is_flower;
    bool is_fruit;

    struct Building* type_building;

    short animal_count;
    
    double size;

    float rotation;
} Branch;

extern Branch* selectedBranch;

extern Branch* branch_list;
extern int branch_count;

extern Product* product_list;
extern int product_count;
